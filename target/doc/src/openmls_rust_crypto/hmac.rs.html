1use hmac::Mac;
2use openmls_traits::types;
3use openmls_traits::types::{CryptoError, HashType};
4use sha2::{Sha256, Sha384, Sha512};
5use tls_codec::SecretVLBytes;
6
7type HmacSha256 = hmac::Hmac<Sha256>;
8type HmacSha384 = hmac::Hmac<Sha384>;
9type HmacSha512 = hmac::Hmac<Sha512>;
10
11macro_rules! hmac_digest {
12    ($hash_t:ty, $key:ident, $message:ident) => {{
13        let mut hmac = <$hash_t>::new_from_slice($key).map_err(|_e| CryptoError::InvalidLength)?;
14        hmac.update($message);
15        hmac.finalize().into_bytes().as_slice().to_vec()
16    }};
17}
18
19pub(crate) fn hmac(
20    hash_type: HashType,
21    key: &[u8],
22    message: &[u8],
23) -> Result<SecretVLBytes, types::CryptoError> {
24    let digest: Vec<u8> = match hash_type {
25        HashType::Sha2_256 => hmac_digest!(HmacSha256, key, message),
26        HashType::Sha2_384 => hmac_digest!(HmacSha384, key, message),
27        HashType::Sha2_512 => hmac_digest!(HmacSha512, key, message),
28    };
29    Ok(SecretVLBytes::new(digest))
30}
