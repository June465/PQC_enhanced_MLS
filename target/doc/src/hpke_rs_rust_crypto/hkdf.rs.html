1use alloc::{vec, vec::Vec};
2
3use hkdf::Hkdf;
4use hpke_rs_crypto::error::Error;
5use sha2::{Sha256, Sha384, Sha512};

7macro_rules! implement_hkdfs {
8    ($name_extract:ident, $name_expand:ident, $hmac_mode:ident, $name:ident) => {
9        pub(crate) fn $name_extract(salt: &[u8], ikm: &[u8]) -> Vec<u8> {
10            Hkdf::<$hmac_mode>::extract(Some(salt), ikm)
11                .0
12                .as_slice()
13                .into()
14        }
15        pub(crate) fn $name_expand(
16            prk: &[u8],
17            info: &[u8],
18            output_size: usize,
19        ) -> Result<Vec<u8>, Error> {
20            let hkdf =
21                Hkdf::<$hmac_mode>::from_prk(prk).map_err(|_| Error::HpkeInvalidOutputLength)?;
22            let mut okm = vec![0u8; output_size];
23            hkdf.expand(info, &mut okm)
24                .map_err(|_| Error::HpkeInvalidOutputLength)?;
25            Ok(okm)
26        }
27    };
28}
