31/// The internal keccak state that can also buffer inputs to absorb.
32/// This is used in the general xof APIs.
33#[cfg_attr(hax, hax_lib::opaque)]
34pub(crate) struct KeccakXofState<
35    const PARALLEL_LANES: usize,
36    const RATE: usize,
37    STATE: KeccakStateItem<PARALLEL_LANES>,
38> {
39    inner: KeccakState<PARALLEL_LANES, STATE>,
40
41    // Buffer inputs on absorb.
42    buf: [[u8; RATE]; PARALLEL_LANES],
43
44    // Buffered length.
45    buf_len: usize,
46
47    // Needs sponge.
48    sponge: bool,
49}
50
51impl<const PARALLEL_LANES: usize, const RATE: usize, STATE: KeccakStateItem<PARALLEL_LANES>>
52    KeccakXofState<PARALLEL_LANES, RATE, STATE>
53{
54    /// An all zero block
55    pub(crate) const fn zero_block() -> [u8; RATE] {
56        [0u8; RATE]
57    }
58
59    /// Generate a new keccak xof state.
60    pub(crate) fn new() -> Self {
61        Self {
62            inner: KeccakState::new(),
63            buf: [Self::zero_block(); PARALLEL_LANES],
64            buf_len: 0,
65            sponge: false,
66        }
67    }
68
69    /// Absorb
70    ///
71    /// This function takes any number of bytes to absorb and buffers if it's not enough.
72    /// The function assumes that all input slices in `blocks` have the same length.
73    ///
74    /// Only a multiple of `RATE` blocks are absorbed.
75    /// For the remaining bytes [`absorb_final`] needs to be called.
76    ///
77    /// This works best with relatively small `inputs`.
78    #[inline(always)]
79    pub(crate) fn absorb(&mut self, inputs: &[&[u8]; PARALLEL_LANES]) {
80        let input_remainder_len = self.absorb_full(inputs);
81
82        // ... buffer the rest if there's not enough input (left).
83        if input_remainder_len > 0 {
84            debug_assert!(
85                self.buf_len == 0  // We consumed everything (or it was empty all along).
86                 || self.buf_len + input_remainder_len <= RATE
87            );
88
89            let input_len = inputs[0].len();
90            for i in 0..PARALLEL_LANES {
91                self.buf[i][self.buf_len..self.buf_len + input_remainder_len]
92                    .copy_from_slice(&inputs[i][input_len - input_remainder_len..]);
93            }
94            self.buf_len += input_remainder_len;
95        }
96    }
97
98    // Note: consciously not inlining this function to avoid using too much stack
99    fn absorb_full(&mut self, inputs: &[&[u8]; PARALLEL_LANES]) -> usize {
100        debug_assert!(PARALLEL_LANES > 0);
101        debug_assert!(self.buf_len < RATE);
102        #[cfg(debug_assertions)]
103        {
104            for block in inputs {
105                debug_assert!(block.len() == inputs[0].len());
106            }
107        }
108
109        // Check if there are buffered bytes to absorb first and consume them.
110        let input_consumed = self.fill_buffer(inputs);
111
112        if input_consumed > 0 {
113            let mut borrowed = [[0u8; RATE].as_slice(); PARALLEL_LANES];
114            // We have a full block in the local buffer now.
115            for i in 0..PARALLEL_LANES {
116                borrowed[i] = &self.buf[i];
117            }
118            STATE::load_block::<RATE>(&mut self.inner.st, &borrowed, 0);
119            keccakf1600(&mut self.inner);
120
121            // "empty" the local buffer
122            self.buf_len = 0;
123        }
124
125        // We only need to consume the rest of the input.
126        let input_to_consume = inputs[0].len() - input_consumed;
127
128        // Consume the (rest of the) input ...
129        let num_blocks = input_to_consume / RATE;
130        let remainder = input_to_consume % RATE;
131        for i in 0..num_blocks {
132            // We only get in here if `input_len / RATE > 0`.
133            STATE::load_block::<RATE>(&mut self.inner.st, &inputs, input_consumed + i * RATE);
134            keccakf1600(&mut self.inner);
135        }
136
137        remainder
138    }
139
140    /// Consume the internal buffer and the required amount of the input to pad to
141    /// `RATE`.
142    ///
143    /// Returns the `consumed` bytes from `inputs` if there's enough buffered
144    /// content to consume, and `0` otherwise.
145    /// If `consumed > 0` is returned, `self.buf` contains a full block to be
146    /// loaded.
147    // Note: consciously not inlining this function to avoid using too much stack
148    fn fill_buffer(&mut self, inputs: &[&[u8]; PARALLEL_LANES]) -> usize {
149        let input_len = inputs[0].len();
150        let mut consumed = 0;
151        if self.buf_len > 0 {
152            // There's something buffered internally to consume.
153            if self.buf_len + input_len >= RATE {
154                // We have enough data when combining the internal buffer and
155                // the input.
156                consumed = RATE - self.buf_len;
157                for i in 0..PARALLEL_LANES {
158                    self.buf[i][self.buf_len..].copy_from_slice(&inputs[i][..consumed]);
159                }
160                self.buf_len += consumed;
161            }
162        }
163        consumed
164    }
165
166    /// Absorb a final block.
167    ///
168    /// The `inputs` block may be empty. Everything in the `inputs` block beyond
169    /// `RATE` bytes is ignored.
170    #[inline(always)]
171    pub(crate) fn absorb_final<const DELIMITER: u8>(&mut self, inputs: &[&[u8]; PARALLEL_LANES]) {
172        let input_remainder_len = self.absorb_full(inputs);
173
174        // Consume the remaining bytes.
175        // This may be in the local buffer or in the input.
176        let input_len = inputs[0].len();
177        let mut blocks = [[0u8; 200]; PARALLEL_LANES];
178        for i in 0..PARALLEL_LANES {
179            if self.buf_len > 0 {
180                blocks[i][0..self.buf_len].copy_from_slice(&self.buf[i][0..self.buf_len]);
181            }
182            if input_remainder_len > 0 {
183                blocks[i][self.buf_len..self.buf_len + input_remainder_len]
184                    .copy_from_slice(&inputs[i][input_len - input_remainder_len..]);
185            }
186            blocks[i][self.buf_len + input_remainder_len] = DELIMITER;
187            blocks[i][RATE - 1] |= 0x80;
188        }
189
190        STATE::load_block_full::<RATE>(&mut self.inner.st, &blocks, 0);
191        keccakf1600(&mut self.inner);
192    }
193
194    /// Squeeze `N` x `LEN` bytes.
195    #[inline(always)]
196    pub(crate) fn squeeze(&mut self, out: [&mut [u8]; PARALLEL_LANES]) {
197        if self.sponge {
198            // If we called `squeeze` before, call f1600 first.
199            // We do it this way around so that we don't call f1600 at the end
200            // when we don't need it.
201            keccakf1600(&mut self.inner);
202        }
203
204        // How many blocks do we need to squeeze out?
205        let out_len = out[0].len();
206        let blocks = out_len / RATE;
207        let last = out_len - (out_len % RATE);
208
209        // Squeeze out one to start with.
210        // XXX: Eurydice does not extract `core::cmp::min`, so we do
211        // this instead. (cf. https://github.com/AeneasVerif/eurydice/issues/49)
212        let mid = if RATE >= out_len { out_len } else { RATE };
213        let (out0, mut out_rest) = STATE::split_at_mut_n(out, mid);
214        STATE::store::<RATE>(&self.inner.st, out0);
215
216        // If we got asked for more than one block, squeeze out more.
217        for _ in 1..blocks {
218            // Here we know that we always have full blocks to write out.
219            let (out0, tmp) = STATE::split_at_mut_n(out_rest, RATE);
220            keccakf1600(&mut self.inner);
221            STATE::store::<RATE>(&self.inner.st, out0);
222            out_rest = tmp;
223        }
224
225        if last < out_len {
226            // Squeeze out the last partial block
227            keccakf1600(&mut self.inner);
228            STATE::store::<RATE>(&self.inner.st, out_rest);
229        }
230
231        self.sponge = true;
232    }
233}

353#[inline(always)]
354pub(crate) fn chi<const N: usize, T: KeccakStateItem<N>>(s: &mut KeccakState<N, T>) {
355    let old = s.clone();
356
357    #[allow(clippy::needless_range_loop)]
358    for i in 0..5 {
359        for j in 0..5 {
360            s.set(
361                i,
362                j,
363                T::and_not_xor(
364                    s.get(i, j),
365                    old.get(i, (j + 2) % 5),
366                    old.get(i, (j + 1) % 5),
367                ),
368            );
369        }
370    }
371}

425#[inline(always)]
426pub(crate) fn absorb_final<
427    const N: usize,
428    T: KeccakStateItem<N>,
429    const RATE: usize,
430    const DELIM: u8,
431>(
432    s: &mut KeccakState<N, T>,
433    last: &[&[u8]; N],
434    start: usize,
435    len: usize,
436) {
437    debug_assert!(N > 0 && len < RATE); // && last[0].len() < RATE
438
439    let mut blocks = [[0u8; 200]; N];
440    for i in 0..N {
441        if len > 0 {
442            blocks[i][0..len].copy_from_slice(&last[i][start..start + len]);
443        }
444        blocks[i][len] = DELIM;
445        blocks[i][RATE - 1] |= 0x80;
446    }
447    T::load_block_full::<RATE>(&mut s.st, &blocks, 0);
448    keccakf1600(s)
449}

532#[inline(always)]
533pub(crate) fn keccak<const N: usize, T: KeccakStateItem<N>, const RATE: usize, const DELIM: u8>(
534    data: &[&[u8]; N],
535    out: [&mut [u8]; N],
536) {
537    let mut s = KeccakState::<N, T>::new();
538    for i in 0..data[0].len() / RATE {
539        // T::slice_n(data, i * RATE, RATE)
540        absorb_block::<N, T, RATE>(&mut s, &data, i * RATE);
541    }
542    let rem = data[0].len() % RATE;
543    // T::slice_n(data, data[0].len() - rem, rem)
544    absorb_final::<N, T, RATE, DELIM>(&mut s, data, data[0].len() - rem, rem);
545
546    let outlen = out[0].len();
547    let blocks = outlen / RATE;
548    let last = outlen - (outlen % RATE);
549
550    if blocks == 0 {
551        squeeze_first_and_last::<N, T, RATE>(&s, out)
552    } else {
553        let (mut o0, mut o1) = T::split_at_mut_n(out, RATE);
554        squeeze_first_block::<N, T, RATE>(&s, &mut o0);
555        for _i in 1..blocks {
556            let (mut o, orest) = T::split_at_mut_n(o1, RATE);
557            squeeze_next_block::<N, T, RATE>(&mut s, &mut o);
558            o1 = orest;
559        }
560        if last < outlen {
561            squeeze_last::<N, T, RATE>(s, o1)
562        }
563    }
564}
