1//! This module describes the storage provider and type traits.
2//! The concept is that the type traits are implemented by OpenMLS, and the storage provider
3//! implements the [`StorageProvider`] trait. The trait mostly defines getters and setters, but
4//! also a few methods that append to lists (which behave similar to setters).
5
6use serde::{de::DeserializeOwned, Serialize};
7/// The storage version used by OpenMLS
8pub const CURRENT_VERSION: u16 = 1;
9
10/// For testing there is a test version defined here.
11///
12/// THIS VERSION MUST NEVER BE USED OUTSIDE OF TESTS.
13#[cfg(any(test, feature = "test-utils"))]
14pub const V_TEST: u16 = u16::MAX;

16/// StorageProvider describes the storage backing OpenMLS and persists the state of OpenMLS groups.
17///
18/// The getters for individual values usually return a `Result<Option<T>, E>`, where `Err(_)`
19/// indicates that some sort of IO or internal error occurred, and `Ok(None)` indicates that no
20/// error occurred, but no value exists.
21/// Many getters for lists return a `Result<Vec<T>, E>`. In this case, if there was no error but
22/// the value doesn't exist, an empty vector should be returned.
23///
24/// Any value that uses the group id as key is required by the group.
25/// Returning `None` or an error for any of them will cause a failure when
26/// loading a group.
27///
28/// More details can be taken from the comments on the respective method.
29pub trait StorageProvider<const VERSION: u16> {
30    /// An opaque error returned by all methods on this trait.
31    type Error: core::fmt::Debug + std::error::Error;
32
33    /// Get the version of this provider.
34    fn version() -> u16 {
35        VERSION
36    }

587// base traits for keys and values
588
589// ANCHOR: key_trait
590/// Key is a trait implemented by all types that serve as a key (in the database sense) to in the
591/// storage. For example, a GroupId is a key to the stored entities for the group with that id.
592/// The point of a key is not to be stored, it's to address something that is stored.
593pub trait Key<const VERSION: u16>: Serialize {}
594// ANCHOR_END: key_trait
595
596// ANCHOR: entity_trait
597/// Entity is a trait implemented by the values being stored.
598pub trait Entity<const VERSION: u16>: Serialize + DeserializeOwned {}
599// ANCHOR_END: entity_trait
600
601impl Entity<CURRENT_VERSION> for bool {}
602impl Entity<CURRENT_VERSION> for u8 {}
603
604// in the following we define specific traits for Keys and Entities. That way
605// we can don't sacrifice type safety in the implementations of the storage provider.
606// note that there are types that are used both as keys and as entities.
607
608// ANCHOR: traits
609/// Each trait in this module corresponds to a type. Some are used as keys, some as
610/// entities, and some both. Therefore, the Key and/or Entity traits also need to be implemented.
611pub mod traits {
612    use super::{Entity, Key};
613
614    // traits for keys, one per data type
615    pub trait GroupId<const VERSION: u16>: Key<VERSION> {}
616    pub trait SignaturePublicKey<const VERSION: u16>: Key<VERSION> {}
617    pub trait HashReference<const VERSION: u16>: Key<VERSION> {}
618    pub trait PskId<const VERSION: u16>: Key<VERSION> {}
619    pub trait EncryptionKey<const VERSION: u16>: Key<VERSION> {}
620    pub trait EpochKey<const VERSION: u16>: Key<VERSION> {}
621
622    // traits for entity, one per type
623    pub trait QueuedProposal<const VERSION: u16>: Entity<VERSION> {}
624    pub trait TreeSync<const VERSION: u16>: Entity<VERSION> {}
625    pub trait GroupContext<const VERSION: u16>: Entity<VERSION> {}
626    pub trait InterimTranscriptHash<const VERSION: u16>: Entity<VERSION> {}
627    pub trait ConfirmationTag<const VERSION: u16>: Entity<VERSION> {}
628    pub trait SignatureKeyPair<const VERSION: u16>: Entity<VERSION> {}
629    pub trait PskBundle<const VERSION: u16>: Entity<VERSION> {}
630    pub trait HpkeKeyPair<const VERSION: u16>: Entity<VERSION> {}
631    pub trait GroupState<const VERSION: u16>: Entity<VERSION> {}
632    pub trait GroupEpochSecrets<const VERSION: u16>: Entity<VERSION> {}
633    pub trait LeafNodeIndex<const VERSION: u16>: Entity<VERSION> {}
634    pub trait MessageSecrets<const VERSION: u16>: Entity<VERSION> {}
