1use std::{
2    collections::HashMap,
3    env,
4    fs::File,
5    io::{BufReader, BufWriter},

9use base64::Engine;
10use serde::{Deserialize, Serialize};
11
12#[derive(Debug, Default, Serialize, Deserialize)]
13struct SerializableKeyStore {
14    values: HashMap<String, String>,
15}
16
17pub fn get_file_path(file_name: &String) -> PathBuf {
18    let tmp_folder = env::temp_dir();
19    tmp_folder.join(file_name)
20}
21
22impl super::MemoryStorage {
23    fn get_file_path(user_name: &str) -> PathBuf {
24        get_file_path(&("openmls_cli_".to_owned() + user_name + "_ks.json"))
25    }
26
27    pub fn save_to_file(&self, output_file: &File) -> Result<(), String> {
28        let writer = BufWriter::new(output_file);
29
30        let mut ser_ks = SerializableKeyStore::default();
31        for (key, value) in &*self.values.read().unwrap() {
32            ser_ks.values.insert(
33                base64::prelude::BASE64_STANDARD.encode(key),
34                base64::prelude::BASE64_STANDARD.encode(value),
35            );
36        }
37
38        match serde_json::to_writer_pretty(writer, &ser_ks) {
39            Ok(()) => Ok(()),
40            Err(e) => Err(e.to_string()),
41        }
42    }
43
44    pub fn save(&self, user_name: String) -> Result<(), String> {
45        let ks_output_path = Self::get_file_path(&user_name);
46
47        match File::create(ks_output_path) {
48            Ok(output_file) => self.save_to_file(&output_file),
49            Err(e) => Err(e.to_string()),
50        }
51    }
52
53    pub fn load_from_file(&mut self, input_file: &File) -> Result<(), String> {
54        // Prepare file reader.
55        let reader = BufReader::new(input_file);
56
57        // Read the JSON contents of the file as an instance of `SerializableKeyStore`.
58        match serde_json::from_reader::<BufReader<&File>, SerializableKeyStore>(reader) {
59            Ok(ser_ks) => {
60                let mut ks_map = self.values.write().unwrap();
61                for (key, value) in ser_ks.values {
62                    ks_map.insert(
63                        base64::prelude::BASE64_STANDARD.decode(key).unwrap(),
64                        base64::prelude::BASE64_STANDARD.decode(value).unwrap(),
65                    );
66                }
67                Ok(())
68            }
69            Err(e) => Err(e.to_string()),
70        }
71    }
72
73    pub fn load(&mut self, user_name: String) -> Result<(), String> {
74        let ks_input_path = Self::get_file_path(&user_name);
75
76        match File::open(ks_input_path) {
77            Ok(input_file) => self.load_from_file(&input_file),
78            Err(e) => Err(e.to_string()),
79        }
80    }
81}
